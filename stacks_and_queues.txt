4. Given a string, reverse it using a stack. For example, the string "Bloc!" should be converted to "!colB".
function reverseString(data){
  
  let stack = [];
  let reverse = '';

  for (var i = 0; i <= data.length; i++)
    stack.push(data[i]);
  

  for (let i = stack.length-1; i >= 0; i--)
    reverse += stack[i];
  return reverse;

}
